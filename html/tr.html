 832    def _wherebuild(self, wheredict, likearr:list=None, verbose:list=None): # ([]string, [])
 833        """
 834         _wherebuild builds wherestrings and fills whereargs.
 835        """
 836        if likearr is None:
 837            likearr = []
 838        if verbose is None:
 839            verbose = []
 840        wherestrs = []
 841        whereargs = []
 842
 843        for key, row in wheredict.items():
 844            #print("key:" + key)
 845
 846            # somehow printing here stops the keys from being iterated in the loop. why?
 847            #print("field: " + row["field"])
 848            
 849            if row["arr"] == None or len(row["arr"]) == 0:
 850                continue
 851
 852            if likearr is not None and key in likearr:
 853                # put in an or-chain of like checks over all elements
 854                s = "(" + self._wherelikes(row["field"], len(row["arr"])) + ")"
 855                wherestrs.append(s)
 856                whereargs.extend(row["arr"])
 857            elif row["arr"] is not None and "type" in row and row["type"] == "date":
 858                if row["arr"] == 'NULL':
 859                    wherestrs.append("(" + row['field'] + " is NULL)")
 860                else:
 861                    s = f"(CAST(" + row['field'] + " as date) between ISNULL(?, " + row['field'] + ") and ISNULL(?, " + row['field'] + "))"
 862                    wherestrs.append(s)
 863                    whereargs.extend(row["arr"])
 864            else:
 865                wherestr = "("
 866                needsor = False
 867                if 'NULL' in row["arr"]:

 878                whereargs.extend(row["arr"])
 879        return (wherestrs, whereargs)

 833    def _wherebuild(self, wheredict, likearr:list=None, verbose:list=None): # ([]string, [])
 834        """
 835         _wherebuild builds wherestrings and fills whereargs.
 836        """
 837        if likearr is None:
 838            likearr = []
 839        if verbose is None:
 840            verbose = []
 841        wherestrs = []
 842        whereargs = []
 843
 844        for key, row in wheredict.items():
 845            #print("key:" + key)
 846
 847            # somehow printing here stops the keys from being iterated in the loop. why?
 848            #print("field: " + row["field"])
 849            
 850            if row["arr"] == None or len(row["arr"]) == 0:
 851                continue
 852
 853            if likearr is not None and key in likearr:
 854                # put in an or-chain of like checks over all elements
 855                s = "(" + self._wherelikes(row["field"], len(row["arr"])) + ")"
 856                wherestrs.append(s)
 857                whereargs.extend(row["arr"])
 858            elif row["arr"] is not None and "type" in row and row["type"] == "date":
 859                if row["arr"] == 'NULL':
 860                    wherestrs.append("(" + row['field'] + " is NULL)")
 861                else:
 862                    s = f"(CAST(" + row['field'] + " as date) between ISNULL(?, " + row['field'] + ") and ISNULL(?, " + row['field'] + "))"
 863                    wherestrs.append(s)
 864                    whereargs.extend(row["arr"])
 865            else:
 866                wherestr = "("
 867                needsor = False
 868                if 'NULL' in row["arr"]:
 869                    row["arr"].remove("NULL")
 870                    wherestr += row["field"] + " is NULL"
 871                    needsor = True
 872                if len(row["arr"]) > 0:
 873                    if needsor:
 874                        wherestr += " or "
 875                    placeholder = traction._sqlinplaceholder(len(row["arr"])) # todo put in package? tr._sqlinplaceholder
 876                    wherestr += row["field"] + " in " + placeholder # e.g. samplelocation.locationpath in (?, ?, ?)
 877                wherestr += ")"
 878                wherestrs.append(wherestr)
 879                whereargs.extend(row["arr"])
 880        return (wherestrs, whereargs)

667        """
